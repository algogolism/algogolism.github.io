    Односвязные списки

    Python

    Я использую Python версии 3.12.

    Начнём с реализации ноды:

    from dataclasses import dataclass
from typing import Optional

@dataclass
class Node:
    value: int
    next: Optional['Node'] = None

    В этом куске экскерментов кода сделано следующее: реализован класс Node, включающий атрибуты value (те самые данные) и next (типа ссылка на следующую ноду).

    Хорошо, возьмёмся за реализацию самого связного списка:

    @dataclass
class LinkedList:
    head: Optional['Node'] = None

    Так, реализован главный класс с атрибутом head, указывающий на начало списка, теперь создадим метод для вставки элемента в конец списка:

    @dataclass
class LinkedList:
    head: Optional['Node'] = None

    def push(self, value: int) -> None:
        new_node = Node(value)

        if not self.head:
            self.head = new_node
            return

        current = self.head

        while current.next:
            current = current.next

        current.next = new_node

    Нууу, пойдём по порядку: сначала мы создаём новый инстанс ноды, далее проверяем, существует ли головной узел, и вставляем сделанную ноду, если нет. Далее мы просто проходим по всем "ссылкам" до ноды, у которой атрибут next равен None. Так мы окажемся в конце списка и вставим новую ноду в этот самый конец.

    Отлично, теперь имплементируем метод pop - удаление элемента из конца списка и возвращение его значения (да, наш односвязный список определяет себя как недостек):

    def pop(self) -> Optional[int]:
        if not self.head:
            return

        if not self.head.next:
            result = self.head.value
            del self.head
            return result

        parent = self.head
        current = self.head.next

        while current.next:
            parent = current
            current = current.next

        result = current.value

        del current
        del parent.next

        return result

    Чтож, всё начинается с той же проверки, что и для push, а далее смотрим, есть ли у корня списка "ребёнок": для удаления последнего элемента односвязного списка нужно удалить не только саму ноду, но и ссылку у его "родителя". Поэтому при проходе через все элементы списка нужно хранить и текущую и предыдущую ноды. Далее точно также проходимся по списку в цикле, сохраняем значение последней ноды, удаляем её и ссылку на неё, возвращаем результат.

    Круто, осталось только реализовать метод для печати всего списка (aka проход по нему):

    def print(self) -> None:
        current = self.head

        while current:
            print(f"{current.value} => ", end="")
            current = current.next

        print("[END]")

    Принцип действия аналогичен, поэтому описывать не буду.

    Вот такая реализация у меня в итоге вышла:

    from dataclasses import dataclass
from typing import Optional

@dataclass
class Node:
    value: int
    next: Optional['Node'] = None

@dataclass
class LinkedList:
    head: Optional['Node'] = None

    # Добавить в конец
    def push(self, value: int) -> None:
        new_node = Node(value)

        if not self.head:
            self.head = new_node
            return

        current = self.head

        while current.next:
            current = current.next

        current.next = new_node

    # Удалить из конца
    def pop(self) -> Optional[int]:
        if not self.head:
            return

        if not self.head.next:
            result = self.head.value
            del self.head
            return result

        parent = self.head
        current = self.head.next

        while current.next:
            parent = current
            current = current.next

        result = current.value

        del current
        del parent.next

        return result

    # Вывести список
    def print(self) -> None:
        current = self.head

        while current:
            print(f"{current.value} => ", end="")
            current = current.next

        print("[END]")

    Давайте протестируем её:

    ll = LinkedList()
ll.push(1)
ll.push(2)
ll.print()
second = ll.pop()
ll.print()
print(f"Выдернуто значение: {second}")

    Вывод:

    1 => 2 => [END]
1 => [END]
Выдернуто значение: 2

    Отлично, всё работает как надо :)

    В итоге я нашёл релизацию через концепцию cons.

    Небольшая справка

    Cons - это наследство, доставшееся от языка Lisp. Грубо говоря, если верить википедии надёжному источнику, это такая функция, которая содержит два значения или два указателя на значения. Она рекурсивна, что делает её удобной для решения нашей задачи. В Rust нет стандартной функции cons, но мы реализуем её при помощи enum.

    Таким образом наш список будет иметь подобное строение:

    (Cons 1 (Cons 2 (Cons 3 Nil)))

    Ну-с, начнём. Сначала зададим enum List:

    enum List {
    Cons(i32, Box<List>),
    Nil,
}

    Тут мы определили нашу недоноду (Cons) и начало списка (Nil).

    Теперь займёмся методами: в силу природы концепции cons подход немного меняется:

    impl List {
    fn new() -> Self {
        List::Nil
    }

    fn push(self, value: i32) -> List {
        List::Cons(value, Box::new(self))
    }
}

    При создании головной ноды - мы просто отдаём Nil. Ну а в методе push мы создаём новую ноду (Cons), которая ссылается на текущую ноду (self).

    Осталось только реализовать методы pop и print:

    fn pop(self) -> (Option<i32>, List) {
    match self {
        List::Cons(value, next) => (Some(value), *next),
        List::Nil => (None, List::Nil),
    }
}

fn print(&self) {
    match self {
        List::Cons(value, next) => {
            print!("{} ", value);
            next.print();
        }
        List::Nil => println!("[START]"),
    }
}

    Я их выделил отдельно: всё дело в том, что они действуют по одной схеме - проходят список при помощи рекурсии. Объяснять тут особо нечего, поэтому вот полный код:

    enum List {
    Cons(i32, Box<List>),
    Nil,
}

impl List {
    fn new() -> Self {
        List::Nil
    }

    fn push(self, value: i32) -> List {
        List::Cons(value, Box::new(self))
    }

    fn pop(self) -> (Option<i32>, List) {
        match self {
            List::Cons(value, next) => (Some(value), *next),
            List::Nil => (None, List::Nil),
        }
    }

    fn print(&self) {
        match self {
            List::Cons(value, next) => {
                print!("{} ", value);
                next.print();
            }
            List::Nil => println!("[START]"),
        }
    }
}

    Тээээкс, а теперь тесты:

    fn main() {
    let mut list = List::new();

    list = list.push(1);
    list = list.push(2);
    list = list.push(3);

    list.print();

    let (value, list) = list.pop();
    println!("Полученное значение: {:?}", value.unwrap());
    list.print();
}

    Выхлоп:

    3 2 1 [START]
Полученное значение: 3
2 1 [START]
